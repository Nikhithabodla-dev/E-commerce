  <script>
    const products = [
      { id: 1, name: "Laptop", price: 50000, image: "https://via.placeholder.com/200?text=Laptop" },
      { id: 2, name: "Smartphone", price: 25000, image: "https://via.placeholder.com/200?text=Smartphone" },
      { id: 3, name: "Headphones", price: 3000, image: "https://via.placeholder.com/200?text=Headphones" },
      { id: 4, name: "Smartwatch", price: 7000, image: "https://via.placeholder.com/200?text=Smartwatch" },
    ];

    let cart = [];

    function displayProducts() {
      const productList = document.getElementById("product-list");
      products.forEach(product => {
        const card = document.createElement("div");
        card.className = "product-card";
        card.innerHTML = `
          <img src="${product.image}" alt="${product.name}">
          <h3>${product.name}</h3>
          <p>â‚¹${product.price}</p>
          <button onclick="addToCart(${product.id})">Add to Cart</button>
        `;
        productList.appendChild(card);
      });
    }

    function addToCart(id) {
      const product = products.find(p => p.id === id);
      cart.push(product);
      updateCart();
    }

    function updateCart() {
      const cartCount = document.getElementById("cart-count");
      const cartItems = document.getElementById("cart-items");
      const totalPrice = document.getElementById("total-price");

      cartCount.textContent = cart.length;
      cartItems.innerHTML = "";

      let total = 0;
      cart.forEach(item => {
        const li = document.createElement("li");
        li.textContent = `${item.name} - â‚¹${item.price}`;
        cartItems.appendChild(li);
        total += item.price;
      });

      totalPrice.textContent = total;
    }

    displayProducts();
  </script>
